31rmqsProducer_t * rmqsProducerCreate(rmqsClientConfiguration_t *ClientConfiguration, const char_t *PublisherReference, PublishResultCallback_t PublishResultCallback)
32{
33 rmqsProducer_t *Producer = (rmqsProducer_t *)rmqsAllocateMemory(sizeof(rmqsProducer_t));
34
35 memset(Producer, 0, sizeof(rmqsProducer_t));
36
37 Producer->Client = rmqsClientCreate(ClientConfiguration, rmqsctProducer, Producer);
38 strncpy(Producer->PublisherReference, PublisherReference, RMQS_MAX_PUBLISHER_REFERENCE_LENGTH);
39 Producer->PollTimer = rmqsTimerCreate();
40 Producer->PublishResultCallback = PublishResultCallback;
41
42 return Producer;
43}

48 rmqsClientDestroy(Producer->Client);
49
50 rmqsFreeMemory((void *)Producer);
51}

53void rmqsProducerPoll(rmqsProducer_t *Producer, const rmqsSocket Socket, uint32_t Timeout)
54{
55 uint32_t WaitMessageTimeout = Timeout;
56 uint32_t Time;
57
58 rmqsTimerStart(Producer->Client->ClientConfiguration->WaitReplyTimer);
59
60 while (rmqsTimerGetTime(Producer->Client->ClientConfiguration->WaitReplyTimer) < Timeout)
61 {
62 rmqsWaitMessage(Producer->Client, Socket, WaitMessageTimeout);
63
64 Time = rmqsTimerGetTime(Producer->Client->ClientConfiguration->WaitReplyTimer);
65
66 WaitMessageTimeout = Timeout - Time;
67 }
68}

70rmqsResponseCode_t rmqsDeclarePublisher(rmqsProducer_t *Producer, const rmqsSocket Socket, const uint8_t PublisherId, const char_t *StreamName)
71{
72 rmqsClient_t *Client = Producer->Client;
73 rmqsClientConfiguration_t *ClientConfiguration = (rmqsClientConfiguration_t *)Client->ClientConfiguration;
74 uint16_t Key = rmqscDeclarePublisher;
75 uint16_t Version = 1;
76
77 rmqsBufferClear(Client->TxQueue, false);
78
79 rmqsAddUInt32ToBuffer(Client->TxQueue, 0, ClientConfiguration->IsLittleEndianMachine); // Size is zero for now
80 rmqsAddUInt16ToBuffer(Client->TxQueue, Key, ClientConfiguration->IsLittleEndianMachine);
81 rmqsAddUInt16ToBuffer(Client->TxQueue, Version, ClientConfiguration->IsLittleEndianMachine);
82 rmqsAddUInt32ToBuffer(Client->TxQueue, ++Client->CorrelationId, ClientConfiguration->IsLittleEndianMachine);
83 rmqsAddUInt8ToBuffer(Client->TxQueue, PublisherId);
84 rmqsAddStringToBuffer(Client->TxQueue, Producer->PublisherReference, ClientConfiguration->IsLittleEndianMachine);
85 rmqsAddStringToBuffer(Client->TxQueue, StreamName, ClientConfiguration->IsLittleEndianMachine);
86
87 //
88 // Moves to the beginning of the stream and writes the total message body size
89 //
90 rmqsBufferMoveTo(Client->TxQueue, 0);
91 rmqsAddUInt32ToBuffer(Client->TxQueue, Client->TxQueue->Size - sizeof(uint32_t), ClientConfiguration->IsLittleEndianMachine);
92
93 rmqsSendMessage(Client, Socket, (const char_t *)Client->TxQueue->Data, Client->TxQueue->Size);
94
95 if (rmqsWaitResponse(Client, Socket, Client->CorrelationId, &Client->Response, 1000))
96 {
97 if (Client->Response.Header.Key != rmqscDeclarePublisher)
98 {
99 return rmqsrWrongReply;
100 }
101
102 return Client->Response.ResponseCode;
103 }
104 else
105 {
106 return rmqsrNoReply;
107 }
108}

148void rmqsPublish(rmqsProducer_t *Producer, const rmqsSocket Socket, const uint8_t PublisherId, rmqsMessage_t *Messages, const size_t MessageCount)
149{
150 rmqsClient_t *Client = Producer->Client;
151 rmqsClientConfiguration_t *ClientConfiguration = (rmqsClientConfiguration_t *)Client->ClientConfiguration;
152 uint16_t Key = rmqscPublish;
153 uint16_t Version = 1;
154 size_t i;
155
156 rmqsBufferClear(Client->TxQueue, false);
157
158 rmqsAddUInt32ToBuffer(Client->TxQueue, 0, ClientConfiguration->IsLittleEndianMachine); // Size is zero for now
159 rmqsAddUInt16ToBuffer(Client->TxQueue, Key, ClientConfiguration->IsLittleEndianMachine);
160 rmqsAddUInt16ToBuffer(Client->TxQueue, Version, ClientConfiguration->IsLittleEndianMachine);
161 rmqsAddUInt8ToBuffer(Client->TxQueue, PublisherId);
162 rmqsAddUInt32ToBuffer(Client->TxQueue, MessageCount, ClientConfiguration->IsLittleEndianMachine); // Message count
163
164 for (i = 0; i < MessageCount; i++)
165 {
166 rmqsAddUInt64ToBuffer(Client->TxQueue, Messages[i].PublishingId, ClientConfiguration->IsLittleEndianMachine);
167 rmqsAddBytesToBuffer(Client->TxQueue, Messages[i].Data, Messages[i].Size, true, ClientConfiguration->IsLittleEndianMachine);
168 }
169
170 //
171 // Moves to the beginning of the stream and writes the total message body size
172 //
173 rmqsBufferMoveTo(Client->TxQueue, 0);
174 rmqsAddUInt32ToBuffer(Client->TxQueue, Client->TxQueue->Size - sizeof(uint32_t), ClientConfiguration->IsLittleEndianMachine);
175
176 rmqsSendMessage(Client, Socket, (const char_t *)Client->TxQueue->Data, Client->TxQueue->Size);
177}

179void rmqsHandlePublishResult(uint16_t Key, rmqsProducer_t *Producer, rmqsBuffer_t *Buffer)
180{
181 char_t *MessagePayload = (char_t *)Buffer->Data + sizeof(rmqsMsgHeader_t);
182 uint8_t *PublisherId;
183 uint32_t *NoOfRecords;
184 uint64_t *PublishingId;
185 uint32_t i;
186 size_t PublishIdCount = 0;
187
188 PublisherId = (uint8_t *)MessagePayload;
189 MessagePayload += sizeof(uint8_t);
190 NoOfRecords = (uint32_t *)MessagePayload;
191 MessagePayload += sizeof(uint32_t);
192
193 if (Producer->Client->ClientConfiguration->IsLittleEndianMachine)
194 {
195 *NoOfRecords = SwapUInt32(*NoOfRecords);
196 }
197
198 if (Key == rmqscPublishConfirm)
199 {
200 for (i = 0; i < *NoOfRecords; i++)
201 {
202 PublishingId = (uint64_t *)MessagePayload;
203
204 if (Producer->Client->ClientConfiguration->IsLittleEndianMachine)
205 {
206 *PublishingId = SwapUInt64(*PublishingId);
207 }
208
209 Producer->PublishingIdResultArray[PublishIdCount++] = *PublishingId;
210
211 if (PublishIdCount == RMQS_PUBLISHING_ID_RESULT_ARRAY_SIZE || i == *NoOfRecords - 1)
212 {
213 Producer->PublishResultCallback(*PublisherId, Producer->PublishingIdResultArray, PublishIdCount, true, 0);
214 PublishIdCount = 0; // Reset the index of the array containing the list of the publish result
215 }
216
217 MessagePayload += sizeof(uint64_t);
218 }
219 }
220 else if (Key == rmqscPublishError)
221 {
222 Key = Key;
223 }
224}
