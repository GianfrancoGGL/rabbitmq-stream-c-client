242inline size_t rmqsAddStringToBuffer(rmqsBuffer_t *Buffer, const char_t *Value, bool_t IsLittleEndianMachine)
243{
244 int16_t StringLen;
245 size_t BytesAdded;
246
247 if (Value == 0 || *Value == 0)
248 {
249 StringLen = RMQS_NULL_STRING_LENGTH;
250 }
251 else
252 {
253 StringLen = (int16_t)strlen(Value);
254 }
255
256 rmqsAddInt16ToBuffer(Buffer, StringLen, IsLittleEndianMachine);
257 BytesAdded = sizeof(StringLen);
258
259 if (StringLen != RMQS_NULL_STRING_LENGTH)
260 {
261 rmqsBufferWrite(Buffer, (void *)Value, StringLen);
262 BytesAdded += StringLen;
263 }
264
265 return BytesAdded;
266}
